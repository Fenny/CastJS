Getting Started

        
        
      
        
// Create new Castjs instance
const cast = new Castjs();

// Optional metadata
var metadata = {
  poster     : 'https://castjs.io/media/poster.jpg',
  title      : 'Sintel',
  description: 'Third Open Movie by Blender Foundation',
  subtitles: [{
      active: true,
      label : 'English',
      src   : 'https://castjs.io/media/english.vtt'
  }, {
      label : 'Spanish',
      src   : 'https://castjs.io/media/spanish.vtt'
  }],
}

// Wait for user interaction
$('#cast').on('click', () => {
  if (cast.available) {
    cast.cast('https://commondatastorage.googleapis.com/gtv-videos-bucket/sample/Sintel.mp4', metadata)
  }
})
